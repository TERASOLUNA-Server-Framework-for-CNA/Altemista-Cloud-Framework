[source,xml]
.new-application-module-web/pom.xml
----
<dependency>
	<groupId>org.altemista.cloudfwk.monitoring</groupId>
	<artifactId>altemista-cloudfwk-web-monitoring</artifactId> <!--1-->
</dependency>
